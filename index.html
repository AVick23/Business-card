def learn(student):
    while not student.is_junior():
        task = get_real_task()
        solution = student.solve(task)
        
        if solution.has_errors():
            # Ошибки = ресурс
            student.analyze_errors()
            student.retry()
        else:
            student.level_up()
    
    return "Оффер получен! 🎉"
